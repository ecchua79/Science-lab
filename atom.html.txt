<script>
    const atoms = ["氢(H)", "氦(He)", "锂(Li)", "铍(Be)", "硼(B)", "碳(C)", "氮(N)", "氧(O)", "氟(F)", "氖(Ne)", "钠(Na)", "镁(Mg)", "铝(Al)", "硅(Si)", "磷(P)", "硫(S)", "氯(Cl)", "氩(Ar)"];
    const mols = [
        { name: "水分子 (H₂O)", formula: { H:2, O:1 } },
        { name: "二氧化碳 (CO₂)", formula: { C:1, O:2 } },
        { name: "氢气 (H₂)", formula: { H:2 } },
        { name: "氧气 (O₂)", formula: { O:2 } },
        { name: "甲烷 (CH₄)", formula: { C:1, H:4 } }
    ];

    let state = {
        startTime: null,
        endTime: null,
        p1: 0,
        p2: 0,
        targetIdx: 0,
        targetMol: null,
        currentAtoms: [],
        usedAtoms: [],
        usedMols: [],
        timerInterval: null
    };

    function startTimer() {
        state.startTime = Date.now();
        state.timerInterval = setInterval(() => {
            const elapsed = Math.floor((Date.now() - state.startTime) / 1000);
            const m = Math.floor(elapsed / 60).toString().padStart(2, '0');
            const s = (elapsed % 60).toString().padStart(2, '0');
            const timeStr = `${m}:${s}`;
            document.getElementById('timer-1').innerText = timeStr;
            document.getElementById('timer-2').innerText = timeStr;
        }, 1000);
    }

    function initAtomLvl() {
        if(!state.startTime) startTimer();
        do { state.targetIdx = Math.floor(Math.random() * 18) + 1; } while (state.usedAtoms.includes(state.targetIdx));
        document.getElementById('target-atom').innerText = atoms[state.targetIdx-1];
        document.getElementById('btn-next-atom').innerText = `验证并继续 (${state.p1}/3)`;
        updateDots('dots-1', state.p1);
        updateAtom();
    }

    function updateAtom() {
        const p = parseInt(document.getElementById('p-slider').value);
        const e = parseInt(document.getElementById('e-slider').value);
        document.getElementById('p-val').innerText = p;
        document.getElementById('e-val').innerText = e;
        const nuc = document.getElementById('nucleus');
        nuc.innerHTML = '';
        for(let i=0; i<p; i++) nuc.innerHTML += '<div class="proton"></div>';
        const orbits = document.getElementById('orbits');
        orbits.innerHTML = '';
        let counts = [Math.min(e, 2), Math.min(Math.max(0, e-2), 8), Math.max(0, e-10)];
        counts.forEach((num, idx) => {
            if(num <= 0) return;
            const r = 55 + idx * 35;
            const orb = document.createElement('div');
            orb.className = 'orbit';
            orb.style.width = orb.style.height = r*2+'px';
            orb.style.left = orb.style.top = `calc(50% - ${r}px)`;
            orbits.appendChild(orb);
            for(let i=0; i<num; i++){
                const el = document.createElement('div');
                el.className = 'electron';
                const ang = (i/num)*Math.PI*2;
                el.style.left = '50%'; el.style.top = '50%';
                el.style.transform = `translate(calc(${Math.cos(ang)*r}px - 4px), calc(${Math.sin(ang)*r}px - 4px))`;
                orbits.appendChild(el);
            }
        });
        document.getElementById('btn-next-atom').disabled = !(p === e && p === state.targetIdx);
    }

    document.getElementById('btn-next-atom').onclick = () => {
        state.usedAtoms.push(state.targetIdx);
        state.p1++;
        if(state.p1 >= 3) {
            document.getElementById('lvl-1').classList.remove('active');
            document.getElementById('lvl-2').classList.add('active');
            initMolLvl();
        } else {
            initAtomLvl();
        }
    };

    function initMolLvl() {
        do { state.targetMol = mols[Math.floor(Math.random() * mols.length)]; } while (state.usedMols.includes(state.targetMol.name));
        document.getElementById('target-mol').innerText = state.targetMol.name;
        document.getElementById('btn-check-mol').innerText = `验证合成 (${state.p2}/3)`;
        updateDots('dots-2', state.p2);
        clearMol();
    }

    function addAtom(t) {
        if(state.currentAtoms.length < 6) {
            state.currentAtoms.push(t);
            renderMol();
        }
    }

    function clearMol() { state.currentAtoms = []; renderMol(); }

    function renderMol() {
        const stage = document.getElementById('mol-stage');
        stage.innerHTML = state.currentAtoms.length ? '' : '<div style="color:#64748b; font-size:0.9rem;">请点击下方原子</div>';
        state.currentAtoms.forEach((t, i) => {
            const div = document.createElement('div');
            div.style.cssText = `position:absolute; width:45px; height:45px; border-radius:50%; display:flex; align-items:center; justify-content:center; font-weight:bold; color:white; box-shadow:0 4px 10px rgba(0,0,0,0.5);`;
            div.style.background = t === 'H' ? 'var(--proton)' : (t === 'O' ? 'var(--primary)' : '#475569');
            const ang = (i/state.currentAtoms.length)*Math.PI*2;
            div.style.transform = `translate(${Math.cos(ang)*65}px, ${Math.sin(ang)*65}px)`;
            div.innerText = t;
            stage.appendChild(div);
        });
    }

    document.getElementById('btn-check-mol').onclick = () => {
        const counts = {};
        state.currentAtoms.forEach(a => counts[a] = (counts[a]||0)+1);
        const target = state.targetMol.formula;
        const allKeys = new Set([...Object.keys(counts), ...Object.keys(target)]);
        let ok = true;
        allKeys.forEach(k => { if(counts[k] !== target[k]) ok = false; });

        if(ok) {
            state.usedMols.push(state.targetMol.name);
            state.p2++;
            if(state.p2 >= 3) {
                state.endTime = Date.now();
                clearInterval(state.timerInterval);
                document.getElementById('name-modal').style.display = 'flex';
            } else {
                initMolLvl();
            }
        } else {
            alert("比例不对！请根据分子式重新配置。");
        }
    };

    function updateDots(id, p) {
        const dots = document.getElementById(id).children;
        for(let i=0; i<3; i++) dots[i].className = i < p ? 'dot active' : 'dot';
    }

    function generateCertificate() {
        const name = document.getElementById('user-name').value;
        if(!name) return alert("写个名字领取你的荣誉吧！");
       
        const now = new Date(state.endTime);
        const totalElapsed = Math.floor((state.endTime - state.startTime) / 1000);
       
        document.getElementById('name-modal').style.display = 'none';
        document.getElementById('display-name').innerText = name;
       
        document.getElementById('cert-date').innerText = "完成日期：" + now.toLocaleDateString('zh-CN', {year:'numeric', month:'long', day:'numeric'});
        document.getElementById('cert-fin-time').innerText = "达成时刻：" + now.toLocaleTimeString('zh-CN', {hour:'2-digit', minute:'2-digit', second:'2-digit'});
        document.getElementById('cert-elapsed').innerText = `通关用时：${Math.floor(totalElapsed/60)}分 ${totalElapsed%60}秒`;
       
        document.getElementById('cert-id').innerText = 'LAB-' + Math.random().toString(36).substr(2, 6).toUpperCase() + '-' + totalElapsed;
        document.getElementById('certificate').style.display = 'flex';
    }

    document.getElementById('p-slider').oninput = document.getElementById('e-slider').oninput = updateAtom;
    window.onload = initAtomLvl;
</script>
